pipeline {
    agent any
 
    stages {
        stage('Build') {
            steps {
                bat 'dotnet build'
            }
        }
        stage('Publish') {
            steps {
                bat 'dotnet publish -c Release -o D:\\Applications\\OfficeBite'
            }
        }
        stage('Deploy') {
            steps {
                powershell '''
                    $source = "D:\\Applications\\GitHubOfficeBiteRepoTest\\bin\\Release\\netcoreapp3.1\\publish"
                    $destination = "D:\\Applications\\OfficeBite"
                    Copy-Item -Path $source\\* -Destination $destination -Recurse -Force
                '''
            }
        }
    }
}
